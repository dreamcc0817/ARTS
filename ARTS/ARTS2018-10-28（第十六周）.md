2-线性结构1 两个有序链表序列的合并 （15 分）
本题要求实现一个函数，将两个链表表示的递增整数序列合并为一个非递减的整数序列

```C
List Merge( List L1, List L2 ){
    List L3=(struct Node *)malloc(sizeof(struct Node));
    List L=L3;
    List p=L1->Next;
    List q=L2->Next;
    while (p && q){
        if (p->Data < q->Data){
            L->Next=p;
            p=p->Next;
        }
        else{
            L->Next=q;
            q=q->Next;
        }
        L=L->Next;
    }
    L->Next=NULL;
    if (p) L->Next=p;
    if (q) L->Next=q;
    L1->Next=NULL;
    L2->Next=NULL;
    return L3;
}
```
